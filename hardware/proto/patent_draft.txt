import numpy as np
import matplotlib.pyplot as plt

# Simulate lathe chatter: random vibrations on a circular surface
theta = np.linspace(0, 2*np.pi, 100)
radius = 1 + 0.05 * np.sin(10 * theta + np.random.randn(100) * 0.1)  # Chatter noise
x = radius * np.cos(theta)
y = radius * np.sin(theta)

plt.plot(x, y, 'b-', label='Chatter Pattern')
plt.axis('equal')
plt.title('Simulated Key Surface')
plt.legend()
plt.show()
